# Mặc định: text auto (Git lưu LF trong repo, Windows checkout CRLF)
* text=auto

# Script/devops nên luôn LF
*.sh   text eol=lf
*.yml  text eol=lf
*.yaml text eol=lf

# Windows script: CRLF
*.bat  text eol=crlf
*.cmd  text eol=crlf
*.ps1  text eol=crlf

# Solution có thể CRLF cho VS
*.sln  text eol=crlf

# Nếu muốn tất cả code .NET là LF (khuyên khi làm đa nền tảng):
*.cs        text eol=lf
*.csproj    text eol=lf
*.props     text eol=lf
*.targets   text eol=lf
*.json      text eol=lf
*.xml       text eol=lf
